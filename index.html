foreach

<script type="text/template" id="users">
    <ul class="users">
        @foreach(users as i:user);
            <li>{{ user.first_name }}</li>
        @endforeach;
    </ul>
</script>

if, else, for, while

<script type="text/template" id="pagination">
    @if(paginate.length);
        <ul class="pagination">
        @for(var i=1; i<=paginate.length; i++);
            <li>{{ i }}</li>
        @endfor;
        </ul>
    @else;
        <div class="warning">page not found.</div>
    @endif;
</script>

include template

<script type="text/template" id="iterator">
    @foreach(users as i:user);
        @include('users', user);
    @endforeach;
</script>

Features

  • Escape String
  • Global Variables
  • For, Foreach and While Loop
  • If, Else and Endif Condition
  • Including Another Template
  • JavaScript Object Notation

Methods

Hide Notification:

if you don't want to see of performance reports use the this method

Thor.HideNotification();

Show Notification:

if you want to see of performance reports use the this method

Thor.ShowNotification();

Set Global Attribute:

if you want to access variables in everywhere use the this method

Thor.SetAttribute('users', []);

// Or

Thor.SetAttribute({
    users : [],
    groups : []
});

Get Global Attribute:

Thor.GetAttribute('attribute');

Load Global Templates:

if you want to use the template in otherwhere use the this method

Thor.LoadTemplate('users');

// Or

Thor.LoadTemplate('users', 'groups');

Examples

Example 1 (Basic Usage)

Template:

<script type="text/template" id="welcome">
    <p>Welcome, {{ username }}!</p>
</script>

JavaScript:

// if use the template this method,
// this template can not be used any more
var Template = new Thor.ReplaceTemplate('welcome', {
    username : 'John Doe'
});

// if want to replace only use the this method
Template.replace();

// if want to source code as string, use the this method
Template.toString();

// if want to use by parent container use the this method
Template.replaceParentNode('div', {
    class : 'welcome-container'
});

Example 2 (Nested Iterate)

Template:

<script type="text/template" id="user">
    <ol class="users">
        @foreach(users as uid:username);
            <li value="{{ uid }}">{{{ username }}}</li>
        @endforeach;
    </ol>
</script>

<script type="text/template" id="users">
    @foreach(groups as gid:params);
    <ul class="groups">
        <li>
            <span>{{{ params.title }}}</span>
            @if(params.users.length);
                @include('user', { users : params.users });
            @endif;
        </li>
    </ul>
    @endforeach;
</script>

JavaScript:

// for include
Thor.LoadTemplate('user');

// render
// if you want to use JSON, must be sent attributes as JSON
new Thor.ReplaceTemplate('users', {
    groups : [
        {
            title : 'Normal Users',
            users : ['Alberta', 'Adrienne', 'Gaby', 'Gaila', 'Talia']
        },
        {
            title : 'Administrator',
            users : ['Tashia', 'Madison', 'Calvin', 'Caine', 'Manton']
        }
    ]
}).replaceParentNode('section', { class : 'users-container' });
